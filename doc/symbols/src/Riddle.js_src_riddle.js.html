 34   function R(first, second) {
 35     if ( typeof first === "string" ) {
 36       if ( typeof second === "undefined" || second instanceof HTMLElement ) {
 37         return $(first, second);
 38       }
 39       else {
 40         throw new Error(readyErr);
 41       }
 42     }
 43     else if ( first instanceof HTMLElement ) {
 44       return wrap(first);
 45     }
 46     else if ( typeof first === "function" ) {
 47       if ( domLoaded ) {
 48         first(r);
 49       }
 50       else {
 51         doc.addEventListener("DOMContentLoaded", function() {
 52           first(r);
 53         }, false);
 54       }
 55     }
 56     else {
 57       throw new Error(readyErr);
 58     }
 59   }

448 /**
449  * send XMLHttpRequest to given URL to get data
450  * @name ajax
451  * @function
452  * @nosideeffects
453  * @memberOf r
454  * @param url {string}
455  * @param success {function(string, Object)}
456  * @param error {?function(Object)}
457  * @param options {?{method: string, header: Object, ctype: string, data: Object}}
458  * @example
459  * r.ajax("http://example.com/people/get", function(data, xhr) {
460  *   r("#people").html(data);
461  * });
462  * @example
463  * r.ajax("http://example.com/articles", function(data, xhr) {
464  *   r("#article").html(data.result);
465  * }, function(xhr) {
466  *   r("#article").html("Oops! Something is wrong!");
467  *   console.dir(xhr);
468  * }, {
469  *   method: "POST",
470  *   data: {
471  *     foo: "bar",
472  *     bar: "baz",
473  *   },
474  *   header: {
475  *     "X-FooBar": "baz"
476  *   },
477  * });
478  */
479   function ajax(url, success, error, options) {
480     var xhr = new XMLHttpRequest(),
481         options = options || {},
482         error = error || function() {},
483         header = options.header || {},
484         ctype = options.ctype || (( options.method === "POST" ) ? "application/x-www-form-urlencoded" : ""),
485         data = options.data || "",
486         key;
487 
488     xhr.onreadystatechange = function() {
489       if ( xhr.readyState === 4 ) {
490         if ( xhr.status >= 200 && xhr.status < 300 ) {
491           success(xhr.responseText, xhr);
492         } else {
493           error(xhr);
494         }
495       }
496     };
497 
498     if ( typeof data === "object" ) {
499       data = encode(data);
500     }
501 
502     xhr.open(options.method || "GET", url, true);
503 
504     if ( ctype ) {
505       xhr.setRequestHeader("Content-Type", ctype);
506     }
507     xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
508 
509     for ( key in header ) {
510       xhr.setRequestHeader(key, header[key]);
511     }
512 
513     xhr.send(data);
514   }
