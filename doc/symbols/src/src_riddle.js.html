 33   function r(first, second) {
 34     if ( typeof first === "string" ) {
 35       return queryWithContext(first, second, "querySelectorAll");
 36     }
 37     else if ( first instanceof HTMLElement ) {
 38       return wrap(elementAsArray(first));
 39     }
 40     else if ( typeof first === "function" ) {
 41       if ( domLoaded ) {
 42         first(r);
 43       }
 44       else {
 45         doc.addEventListener("DOMContentLoaded", function() {
 46           first(r);
 47         }, false);
 48       }
 49     }
 50   }

 52   function queryWithContext(selector, context, functionName) {
 53     if ( typeof context === "undefined" || context instanceof HTMLElement ) {
 54       return wrap(elementAsArray((context || doc)[functionName](selector)));
 55     }
 56     else if ( context.__proto__ === r.fn ) {
 57       var selecteds = context.map(function(el) { return elementAsArray(el[functionName](selector)); });
 58       return wrap(selecteds.reduce(function(all, cur) { return all.concat(cur); }, []));
 59     }
 60   }

191     var outers;
192 
193     if ( typeof item === "undefined" ) {
194       if ( this.length === 1 ) {
195         return this[0].innerHTML;
196       }
197       else {
198         return this.pluck("innerHTML").map(function(html) { return html; });

204       }

207       }
208       else if ( item.__proto__ === r.fn ) {
209         outers = item.map(function(el) { return el.outerHTML; }).join("");
210         this.forEach(function(elem) {
211           elem.innerHTML = outers;
212         });
213       }
214     }
215     return this;

451 /**
452  * bind callback function to elements
453  * @name bind
454  * @function
455  * @memberOf r.fn
456  * @param event {string}
457  * @param callback {function(e: Object)}
458  * @param useCapture {?boolean}
459  * @example
460  * r("#button").bind("click", function(e) {
461  *   alert("button clicked on" + e.target.tagName);
462  * });
463  */
464   function bind(event, callback, useCapture) {
465     this.forEach(function(elem) {
466       var id = getNodeId(elem),
467           bounds = listeners[id] || (listeners[id] = []);
468       bounds.push( { event: event, callback: callback, index: bounds.length } );
469       elem.addEventListener(event, callback, useCapture || false);
470     });
471     return this;
472   }

474 /**
475  * unbind alreaedy-bound callback function from elements
476  * @name unbind
477  * @function
478  * @memberOf r.fn
479  * @param event {string}
480  * @example
481  * r("#button").unbind("click");
482  */
483   function unbind(event) {
484     this.forEach(function(elem) {
485       var id = getNodeId(elem),
486           bounds = event ? findBoundsByEvent(listeners[id], event) : listeners[id];
487       bounds && bounds.forEach(function(bound) {
488         delete bounds[bound.index];
489         elem.removeEventListener(bound.event, bound.callback, false);
490       });
491     });
492     return this;
493   }

508 /**
509  * send XMLHttpRequest to given URL to get data
510  * @name ajax
511  * @memberOf r
512  * @function
513  * @param url {string}
514  * @param success {function(string, Object)}
515  * @param error {?function(Object)}
516  * @param options {?{method: string, header: Object, ctype: string, data: Object}}
517  * @example
518  * r.ajax("http://example.com/people/get", function(data, xhr) {
519  *   r("#people").html(data);
520  * });
521  * @example
522  * r.ajax("http://example.com/articles", function(data, xhr) {
523  *   r("#article").html(data.result);
524  * }, function(xhr) {
525  *   r("#article").html("Oops! Something is wrong!");
526  *   console.dir(xhr);
527  * }, {
528  *   method: "POST",
529  *   data: {
530  *     foo: "bar",
531  *     bar: "baz",
532  *   },
533  *   header: {
534  *     "X-FooBar": "baz"
535  *   },
536  * });
537  */
538   function ajax(url, success, error, options) {
539     var xhr = new XMLHttpRequest(),
540         options = options || {},
541         error = error || function() {},
542         header = options.header || {},
543         ctype = options.ctype || (( options.method === "POST" ) ? "application/x-www-form-urlencoded" : ""),
544         data = options.data || "",
545         key;
546 
547     xhr.onreadystatechange = function() {
548       if ( xhr.readyState === 4 ) {
549         if ( xhr.status >= 200 && xhr.status < 300 ) {
550           success(xhr.responseText, xhr);
551         } else {
552           error(xhr);
553         }
554       }
555     };
556 
557     if ( typeof data === "object" ) {
558       data = encode(data);
559     }
560 
561     xhr.open(options.method || "GET", url, true);
562 
563     if ( ctype ) {
564       xhr.setRequestHeader("Content-Type", ctype);
565     }
566     xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
567 
568     for ( key in header ) {
569       xhr.setRequestHeader(key, header[key]);
570     }
571 
572     xhr.send(data);
573   }
