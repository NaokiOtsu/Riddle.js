 12     /**
 13      * <p> Select HTMLElements, wait DOMContentLoaded or wrap HTMLElement with r.fn <br /> Usages are:
 14      * <ul>
 15      * <li>Select and wrap HTML Elements with r(selector: String, [context: HTMLElement]).
 16      * <li>Wait DOMContentLoaded with r(func: Function). <br/>
 17      * <li>Just to wrap HTMLElement with r(elem: HTMLElement)
 18      * <ul/>
 19      * @name r
 20      * @namespace
 21      * @function
 22      * @param first {(string|function|HTMLElement)}
 23      * @param second {?(HTMLElement|NodeArray)}
 24      * @return {NodeArray} On Selector or wrapper usage
 25      * @example
 26      * var elementsById = r("#id");
 27      * var elementsByClass = r(".class");
 28      * var elementsByTag = r("tag");
 29      * @example
 30      * r(function() {
 31      *   // put initialization process here, which you want to run after DOMContentLoaded event
 32      * });
 33      * @example
 34      * r("#id").bind(function(e) {
 35      *   var wrapped = r(e.target);
 36      * });
 37     */
 38     function r(first, second) {
 39         if ( typeof first === "string" ) {
 40             return queryWithContext(first, second, "querySelectorAll");
 41         }
 42         else if ( first instanceof Array ) {
 43             return wrap(first);
 44         }
 45         else if ( typeof first.addEventListener === "function" ) {
 46             return wrap(elementAsArray(first));
 47         }
 48         else if ( typeof first === "function" ) {
 49             if ( domLoaded ) {
 50                 first(r);
 51             }
 52             else {
 53                 doc.addEventListener("DOMContentLoaded", function() {
 54                     first(r);
 55                 }, false);
 56             }
 57         }
 58     }

 60     function queryWithContext(selector, context, functionName) {
 61         var result = [];
 62 
 63         if ( context === void 0 || context instanceof HTMLElement ) {
 64             return wrap(elementAsArray((context || doc)[functionName](selector)));
 65         }
 66         else if ( r.isR(context) ) {
 67             context.map(function(el) {
 68                 return elementAsArray(el[functionName](selector));
 69             }).forEach(function(array) {
 70                 array.forEach(function(el) {
 71                     result.push(el);
 72                 });
 73             });
 74             return wrap(result);
 75         }
 76     }

 78     function elementAsArray(el) {
 79         if ( el === null || el === void 0 ) {
 80             return [];
 81         }
 82         else if ( el instanceof HTMLElement ) {
 83             return [el];
 84         }
 85         else if ( typeof el.length === "number" && typeof el.item === "function" ) {
 86             return toArray.call(el);
 87         }

529                 });

533                 bounds = event ? findBoundsByEvent(listeners[id], event) : listeners[id];

606             method = options.method || "GET",

613             if ( xhr.readyState === 4 ) {
614                 if ( xhr.status >= 200 && xhr.status < 300 ) {
615                     success(xhr.responseText, xhr);

617                     error(xhr);
618                 }
619             }
620         };

624         }

629             xhr.setRequestHeader("Content-Type", ctype);
630         }
631         xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
632 
633         for ( key in header ) {
