 34     if ( typeof first === "string" ) {
 35       return queryWithContext(first, second, "querySelectorAll");
 36     }
 37     else if ( first instanceof HTMLElement ) {
 38       return wrap(elementAsArray(first));
 39     }
 40     else if ( typeof first === "function" ) {
 41       if ( domLoaded ) {
 42         first(r);
 43       }
 44       else {
 45         doc.addEventListener("DOMContentLoaded", function() {

 47         }, false);
 48       }
 49     }

206     var outers;
207 
208     if ( typeof item === "undefined" ) {
209       if ( this.length === 1 ) {

211       }

222       }

495 /**
496  * bind callback function to elements
497  * @name bind
498  * @function
499  * @memberOf r.fn
500  * @param event {string}
501  * @param callback {function(e: Object)}
502  * @param useCapture {?boolean}
503  * @example
504  * r("#button").bind("click", function(e) {
505  *   alert("button clicked on" + e.target.tagName);
506  * });
507  */
508   function bind(event, callback, useCapture) {
509     this.forEach(function(elem) {
510       var id = getNodeId(elem),
511           bounds = listeners[id] || (listeners[id] = []);
512       bounds.push( { event: event, callback: callback, index: bounds.length } );
513       elem.addEventListener(event, callback, useCapture || false);
514     });
515     return this;
516   }

518 /**
519  * unbind alreaedy-bound callback function from elements
520  * @name unbind
521  * @function
522  * @memberOf r.fn
523  * @param event {string}
524  * @example
525  * r("#button").unbind("click");

527   function unbind(event) {
528     this.forEach(function(elem) {
529       var id = getNodeId(elem),
530           bounds = event ? findBoundsByEvent(listeners[id], event) : listeners[id];
531       bounds && bounds.forEach(function(bound) {
532         delete bounds[bound.index];
533         elem.removeEventListener(bound.event, bound.callback, false);
534       });
535     });
536     return this;
537   }

552 /**
553  * send XMLHttpRequest to given URL to get data
554  * @name ajax
555  * @memberOf r
556  * @function
557  * @param url {string}
558  * @param success {function(string, Object)}
559  * @param error {?function(Object)}
560  * @param options {?{method: string, header: Object, ctype: string, data: Object}}
561  * @example
562  * r.ajax("http://example.com/people/get", function(data, xhr) {
563  *   r("#people").html(data);
564  * });
565  * @example
566  * r.ajax("http://example.com/articles", function(data, xhr) {
567  *   r("#article").html(data.result);
568  * }, function(xhr) {
569  *   r("#article").html("Oops! Something is wrong!");
570  *   console.dir(xhr);
571  * }, {
572  *   method: "POST",
573  *   data: {
574  *     foo: "bar",
575  *     bar: "baz",
576  *   },
577  *   header: {
578  *     "X-FooBar": "baz"
579  *   },
580  * });
581  */
582   function ajax(url, success, error, options) {
583     var xhr = new XMLHttpRequest(),
584         options = options || {},
585         error = error || function() {},
586         header = options.header || {},
587         ctype = options.ctype || (( options.method === "POST" ) ? "application/x-www-form-urlencoded" : ""),
588         data = options.data || "",
589         key;
590 
591     xhr.onreadystatechange = function() {
592       if ( xhr.readyState === 4 ) {
593         if ( xhr.status >= 200 && xhr.status < 300 ) {
594           success(xhr.responseText, xhr);
595         } else {
596           error(xhr);
597         }
598       }
599     };
600 
601     if ( typeof data === "object" ) {
602       data = encode(data);
603     }
604 
605     xhr.open(options.method || "GET", url, true);
606 
607     if ( ctype ) {
608       xhr.setRequestHeader("Content-Type", ctype);
609     }
610     xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
611 
612     for ( key in header ) {
613       xhr.setRequestHeader(key, header[key]);
614     }
615 
616     xhr.send(data);
617   }
