 24     function single(wrapped, properties, options) {
 25         var dur = 0.3,
 26             ease = "ease-in-out",
 27             transNum,
 28             endCall = 0,
 29             cssParams = {};
 30 
 31         if ( properties instanceof Array ) {
 32             cssParams["-webkit-transform"] = properties.join(" ");
 33             transNum = 1;
 34         }
 35         else if ( typeof properties === "object" ) {
 36             transNum = copy(properties, cssParams);
 37         }
 38 
 39         if ( typeof options === "object" ) {
 40 
 41             if ( typeof options.duration === "number" || typeof options.duration === "string" ) {
 42                 dur = options.duration;
 43             }
 44 
 45             if ( typeof options.easing === "string" ) {
 46                 ease = options.easing;
 47             }
 48 
 49             if ( typeof options.callback === "function" ) {
 50                 wrapped.bind("webkitTransitionEnd", function(evt) {
 51                     ++endCall;
 52                     if ( transNum === endCall ) {
 53                         options.callback(evt);
 54                         wrapped.unbind("webkitTransitionEnd");
 55                     }
 56                 });
 57             }
 58 
 59         }
 60 
 61         cssParams["-webkit-transition"] = dur + "s " + ease;
 62 
 63         wrapped.css(cssParams);
 64     }

 66     function copy(from, to) {
 67         var key, count = 0;
 68         for ( key in from ) {
 69             to[key] = from[key];
 70             count++;
 71         }

 73     }
