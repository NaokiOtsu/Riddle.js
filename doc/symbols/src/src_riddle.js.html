  8     /**
  9      * <p> Select HTMLElements, wait DOMContentLoaded or wrap HTMLElement with r.fn <br /> Usages are:
 10      * <ul>
 11      * <li>Select and wrap HTML Elements with r(selector: String, [context: HTMLElement]).
 12      * <li>Wait DOMContentLoaded with r(func: Function). <br/>
 13      * <li>Just to wrap HTMLElement with r(elem: HTMLElement)
 14      * <ul/>
 15      * @name r
 16      * @namespace
 17      * @function
 18      * @param first {(string|function|HTMLElement)}
 19      * @param second {?(HTMLElement|NodeArray)}
 20      * @return {NodeArray} On Selector or wrapper usage
 21      * @example
 22      * var elementsById = r("#id");
 23      * var elementsByClass = r(".class");
 24      * var elementsByTag = r("tag");
 25      * @example
 26      * r(function() {
 27      *   // put initialization process here, which you want to run after DOMContentLoaded event
 28      * });
 29      * @example
 30      * r("#id").bind(function(e) {
 31      *   var wrapped = r(e.target);
 32      * });
 33     */
 34     function r(first, second) {
 35         if ( typeof first === "string" ) {
 36             return queryWithContext(first, second, "querySelectorAll");
 37         }
 38         else if ( first instanceof HTMLElement ) {
 39             return wrap(elementAsArray(first));
 40         }
 41         else if ( typeof first === "function" ) {
 42             if ( domLoaded ) {
 43                 first(r);
 44             }
 45             else {
 46                 doc.addEventListener("DOMContentLoaded", function() {
 47                     first(r);
 48                 }, false);
 49             }
 50         }
 51     }

 53     function queryWithContext(selector, context, functionName) {
 54         if ( typeof context === "undefined" || context instanceof HTMLElement ) {
 55             return wrap(elementAsArray((context || doc)[functionName](selector)));
 56         }
 57         else if ( context.__proto__ === r.fn ) {
 58             var result = [];
 59             context.map(function(el) {
 60                 return elementAsArray(el[functionName](selector));
 61             }).forEach(function(array) {
 62                 array.forEach(function(el) {
 63                     result.push(el);
 64                 });
 65             });
 66             return wrap(result);
 67         }
 68     }

 71         if ( el === null ) {

 73         }

 75             return [el];
 76         }
 77         else if ( typeof el.length === "number" && typeof el.item === "function" ) {
 78             return toArray.call(el);
 79         }

196      * <p> Get/Set innerHTML of elements </p>
197      * <ul>
198      * <li> html(): returns html: String if selector has just one element
199      * <li> html(): returns htmls: Array[String] if selector has more than two elements
200      * <li> html(str): set string as innerHTML for all elements
201      * <li> html(elem): set HTMLElement as innerHTML for all elements
202      * <li> html(nodeArray): set NodeArray as innerHTML for all elements
203      * <ul/>
204      * @name html
205      * @function
206      * @memberOf r.fn
207      * @param html {(string|HTMLElement|NodeArray)}
208      * @return {(string|Array.<string>)}
209      * @example
210      * var story = r("p#story").html();
211      * @example
212      * var colors = r("li.colors").html();
213      * @example
214      * r("li.colors").html("black");
215      * @example
216      * r("li.colors").html(document.getElementById("#my-color"));
217      * @example
218      * r("#story").html(r("li#stories"));
219     */
220     function html(item) {
221         var outers;
222 
223         if ( typeof item === "undefined" ) {
224             if ( this.length === 1 ) {
225                 return this[0].innerHTML;
226             }
227             else {
228                 return this.pluck("innerHTML").map(function(html) {
229                     return html;
230                 });
231             }
232         }
233         else {
234             if ( typeof item === "string" ) {
235                 this.forEach(function(elem) {
236                     elem.innerHTML = item;
237                 });
238             }
239             else if ( item instanceof HTMLElement ) {
240                 this.forEach(function(elem) {
241                     elem.innerHTML = item.outerHTML;
242                 });
243             }
244             else if ( item.__proto__ === r.fn ) {
245                 outers = item.map(function(el) {
246                     return el.outerHTML;
247                 }).join("");
248 
249                 this.forEach(function(elem) {
250                     elem.innerHTML = outers;
251                 });
252             }
253         }
254         return this;
255     }

342      * <p> Get/Set attribute(s) of elements </p>
343      * <ul>
344      * <li> attr(name): returns attribute: String if selector has just one element
345      * <li> attr(name): returns attributes: Array[String] if selector has more than two elements
346      * <li> attr(name, value): set value to element's attribute. Remove attribute if value === null
347      * <li> attr(hash): set values to element's attribute. Remove attribute if hash.value === null
348      * <ul/>
349      * @name attr
350      * @function
351      * @memberOf r.fn
352      * @param first {(string|Object)}
353      * @param second {?string}
354      * @return {(string|Array.<string>)}
355      * @example
356      * var value = r("#age").attr("value");
357      * @example
358      * var values = r("option.age").attr("value");
359      * @example
360      * r(".links-change").attr("href", "http://example.com");
361      * @example
362      * r(".links-change").attr("href", null);
363      * @example
364      * r(".links-change").attr( { href: "http://example.com", target: "_blank" } );
365      * @example
366      * r(".links-change").attr( { href: null } );
367     */
368     function attr(first, second) {
369         if ( typeof first === "string" ) {
370             if ( typeof second === "undefined" ) {
371                 if ( this.length === 1 ) {
372                     return this[0].getAttribute(first);
373                 }
374                 else {
375                     return this.invoke("getAttribute", first);
376                 }
377             }
378             else {
379                 if ( second === null ) {
380                     this.invoke("removeAttribute", first);
381                 }
382                 else {
383                     this.invoke("setAttribute", first, String(second));
384                 }
385             }
386         }
387         else if ( typeof first === "object" ) {
388             this.forEach(function(elem) {
389                 for ( var k in first ) {
390                     if ( first[k] === null ) {
391                         elem.removeAttribute(k);
392                     }
393                     else {
394                         elem.setAttribute(k, String(first[k]));
395                     }
396                 }
397             });
398         }
399         return this;
400     }

403      * <p> Get/Set css property(properties) of elements </p>
404      * <ul>
405      * <li> css(key): returns css style value: String if selector has just one element
406      * <li> css(key): returns css style values: Array[String] if selector has more than two elements
407      * <li> css(key, value): set css style value. Remove css attribute if value === null
408      * <li> css(hash): set css style values. Remove css attribute if hash.value === null
409      * <ul/>
410      * @name css
411      * @function
412      * @memberOf r.fn
413      * @param first {(string|Object)}
414      * @param second {?string}
415      * @return {(string|Array.<string>)}
416      * @example
417      * var bodyHeight = r("body").css("height");
418      * @example
419      * var listHeights = r("li.familiar").css("height");
420      * @example
421      * r("#hyde").css("height", "156px");
422      * @example
423      * r("#hyde").css("height", null);

425      * r(".monster").css( { visibility: "visible", background-color: "red" } );
426      * @example
427      * r(".monster").css( { visibility: null } );
428     */
429     function css(first, second) {
430         if ( typeof first === "string" ) {
431             if ( typeof second === "undefined" ) {
432                 if ( this.length === 1 ) {
433                     return getComputedStyle(this[0], "").getPropertyValue(first);
434                 }
435                 else {
436                     return this.map(function(elem) {
437                         return getComputedStyle(elem, "").getPropertyValue(first);
438                     });
439                 }
440             }
441             else {
442                 if ( second === null ) {
443                     this.forEach(function(elem) {
444                         elem.style.removeProperty(first);
445                     });
446                 }
447                 else {
448                     this.forEach(function(elem) {
449                         elem.style.setProperty(first, second, "");
450                     });
451                 }
452             }
453         }
454         else if ( typeof first === "object" ) {
455             this.forEach(function(elem) {
456                 for ( var key in first ) {
457                     if ( first[key] === null ) {
458                         elem.style.removeProperty(key);
459                     }
460                     else {
461                         elem.style.setProperty(key, first[key], "");
462                     }
463                 }
464             });
465         }
466         return this;
467     }

545     /**
546      * bind callback function to elements
547      * @name bind
548      * @function
549      * @memberOf r.fn
550      * @param event {string}
551      * @param callback {function(e: Object)}
552      * @param useCapture {?boolean}
553      * @example
554      * r("#button").bind("click", function(e) {
555      *   alert("button clicked on" + e.target.tagName);
556      * });
557     */
558     function bind(event, callback, useCapture) {
559         this.forEach(function(elem) {
560             var id = getNodeId(elem),
561             bounds = listeners[id] || (listeners[id] = []);
562             bounds.push( { event: event, callback: callback, index: bounds.length } );
563             elem.addEventListener(event, callback, useCapture || false);
564         });
565         return this;
566     }

568     /**
569      * unbind alreaedy-bound callback function from elements
570      * @name unbind
571      * @function
572      * @memberOf r.fn
573      * @param event {string}
574      * @example
575      * r("#button").unbind("click");
576     */
577     function unbind(event) {
578         this.forEach(function(elem) {
579             var id = getNodeId(elem),
580             bounds = event ? findBoundsByEvent(listeners[id], event) : listeners[id];
581             bounds && bounds.forEach(function(bound) {
582                 delete bounds[bound.index];
583                 elem.removeEventListener(bound.event, bound.callback, false);
584             });
585         });
586         return this;
587     }

602     /**
603      * send XMLHttpRequest to given URL to get data
604      * @name ajax
605      * @memberOf r
606      * @function
607      * @param url {string}
608      * @param success {function(string, Object)}
609      * @param error {?function(Object)}
610      * @param options {?{method: string, header: Object, ctype: string, data: Object}}
611      * @example
612      * r.ajax("http://example.com/people/get", function(data, xhr) {
613      *   r("#people").html(data);
614      * });
615      * @example
616      * r.ajax("http://example.com/articles", function(data, xhr) {
617      *   r("#article").html(data.result);
618      * }, function(xhr) {
619      *   r("#article").html("Oops! Something is wrong!");
620      *   console.dir(xhr);
621      * }, {
622      *   method: "POST",
623      *   data: {
624      *     foo: "bar",
625      *     bar: "baz",
626      *   },
627      *   header: {
628      *     "X-FooBar": "baz"
629      *   },
630      * });
631     */
632     function ajax(url, success, error, options) {
633         var xhr = new XMLHttpRequest(),
634             options = options || {},
635             success = success || function() {},
636             error = error || function() {},
637             method = options.method || "GET",
638             header = options.header || {},
639             ctype = options.ctype || (( method === "POST" ) ? "application/x-www-form-urlencoded" : ""),
640             data = options.data || "",
641             key;
642 
643         xhr.onreadystatechange = function() {
644             if ( xhr.readyState === 4 ) {
645                 if ( xhr.status >= 200 && xhr.status < 300 ) {
646                     success(xhr.responseText, xhr);
647                 } else {
648                     error(xhr);
649                 }
650             }
651         };
652 
653         if ( typeof data === "object" ) {
654             data = encode(data);
655         }
656 
657         xhr.open(method, url, true);
658 
659         if ( ctype ) {
660             xhr.setRequestHeader("Content-Type", ctype);
661         }
662         xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
663 
664         for ( key in header ) {
665             xhr.setRequestHeader(key, header[key]);
666         }
667 
668         xhr.send(data);
669     }
